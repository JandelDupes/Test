-- GAG Dark Dev Scripts GUI by RedOwl
local Players = game:GetService("Players")
local Player = Players.LocalPlayer
local PlayerGui = Player:WaitForChild("PlayerGui")

local ScreenGui = Instance.new("ScreenGui")
local MainFrame = Instance.new("Frame")
local UICorner = Instance.new("UICorner")
local Title = Instance.new("TextLabel")
local Button = Instance.new("TextButton")
local Drag = Instance.new("Frame")

ScreenGui.Name = "GAGDarkDevGUI"
ScreenGui.ResetOnSpawn = false
ScreenGui.Parent = PlayerGui

MainFrame.Name = "MainFrame"
MainFrame.Parent = ScreenGui
MainFrame.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
MainFrame.Position = UDim2.new(0.3, 0, 0.3, 0)
MainFrame.Size = UDim2.new(0, 280, 0, 140)

UICorner.CornerRadius = UDim.new(0, 10)
UICorner.Parent = MainFrame

-- Drag functionality
Drag.Name = "Drag"
Drag.Parent = MainFrame
Drag.BackgroundTransparency = 1
Drag.Size = UDim2.new(1, 0, 0.2, 0)

local dragging, dragInput, dragStart, startPos
Drag.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = true
		dragStart = input.Position
		startPos = MainFrame.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then dragging = false end
		end)
	end
end)
Drag.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement then dragInput = input end
end)
game:GetService("UserInputService").InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		local delta = input.Position - dragStart
		MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X,
									   startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)

-- Title
Title.Name = "Title"
Title.Parent = MainFrame
Title.BackgroundTransparency = 1
Title.Position = UDim2.new(0, 0, 0, 5)
Title.Size = UDim2.new(1, 0, 0.2, 0)
Title.Text = "GAG Dark Dev Scripts"
Title.Font = Enum.Font.GothamBold
Title.TextColor3 = Color3.fromRGB(255, 255, 255)
Title.TextSize = 18

-- Button
Button.Name = "ChangePetCD"
Button.Parent = MainFrame
Button.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
Button.Position = UDim2.new(0.15, 0, 0.45, 0)
Button.Size = UDim2.new(0.7, 0, 0.3, 0)
Button.Font = Enum.Font.GothamSemibold
Button.Text = "Change Pet CD to 15 Seconds"
Button.TextColor3 = Color3.fromRGB(255, 255, 255)
Button.TextSize = 14

local btnCorner = Instance.new("UICorner", Button)
btnCorner.CornerRadius = UDim.new(0, 6)

-- 🔧 Actual Pet Cooldown Spoof Logic
Button.MouseButton1Click:Connect(function()
	local changed = false
	for _, gui in ipairs(PlayerGui:GetDescendants()) do
		if gui:IsA("TextLabel") then
			-- Active Pets (13:02m -> 15s)
			if string.match(gui.Text, "^%d+:%d%d[m]?$") then
				gui.Visible = false
				local fakeCD = Instance.new("TextLabel")
				fakeCD.Size = gui.Size
				fakeCD.Position = gui.Position
				fakeCD.Font = gui.Font
				fakeCD.BackgroundTransparency = 1
				fakeCD.TextColor3 = Color3.fromRGB(255, 255, 0)
				fakeCD.TextSize = gui.TextSize
				fakeCD.Text = "15s"
				fakeCD.TextScaled = gui.TextScaled
				fakeCD.Parent = gui.Parent
				changed = true
			end

			-- Pet Info Section
			if string.match(gui.Text, "opens its frills and spits out") then
				gui.Visible = false
				local fakeInfo = Instance.new("TextLabel")
				fakeInfo.Size = gui.Size
				fakeInfo.Position = gui.Position
				fakeInfo.Font = gui.Font
				fakeInfo.BackgroundTransparency = 1
				fakeInfo.TextColor3 = Color3.fromRGB(200, 255, 255)
				fakeInfo.TextSize = gui.TextSize
				fakeInfo.TextWrapped = gui.TextWrapped
				fakeInfo.TextXAlignment = gui.TextXAlignment
				fakeInfo.Text = "Every 0.15m opens its frills and spits out venom! (Cooldown spoofed)"
				fakeInfo.Parent = gui.Parent
				changed = true
			end
		end
	end

	if changed then
		Button.Text = "Cooldown Spoofed ✓"
	end
end)
